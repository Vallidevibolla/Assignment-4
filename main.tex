\section{Question No.2.1}
Check which of the following are solutions of the equation
\begin{align}
\myvec{1&-2}x=4\\
(a)\myvec{0\\2} (b)\myvec{4\\0} (c)\myvec{2\\0} (d)\myvec{\sqrt{2}\\4\sqrt{2}} (e)\myvec{1\\1}
\end{align}
\section{Solution}
Given 
\myvec{1&-2}x=4\\
let 'y' be the solution then equation be 
\begin{equation} \label{eq:1}
\boxed{y=\myvec{1&-2}x}
\end{equation}

\begin{align}
(a) Substitute x=\myvec{0\\2}in \eqref{eq:1}\\
          y=\myvec{1&-2}\myvec{0\\2}\\
          \boxed{y=-4}\\
(b) Substitute x=\myvec{4\\0}in \eqref{eq:1}\\
          y=\myvec{1&-2}\myvec{4\\0}\\
          \boxed{y=4}\\
(c) Substitute x=\myvec{2\\0}in \eqref{eq:1}\\
          y=\myvec{1&-2}\myvec{2\\0}\\
          \boxed{y=2}\\
(d) Substitute x=\myvec{\sqrt{2}\\4\sqrt{2}}in \eqref{eq:1}\\
          y=\myvec{1&-2}\myvec{\sqrt{2}\\4\sqrt{2}}\\
          \boxed{y=-7\sqrt{2}}\\
(e) Substitute x=\myvec{1\\1}in \eqref{eq:1}\\
          y=\myvec{1&-2}\myvec{1\\1}\\
          \boxed{y=-1}
\end{align}
$\therefore$ 
x=\myvec{4\\0} is the solution of the equation \myvec{1&-2}x=4
